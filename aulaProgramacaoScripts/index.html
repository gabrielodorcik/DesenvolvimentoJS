Exercicio 1

Escreva uma função em JavaScript que realize a conversão de uma temperatura fornecida em graus Fahrenheit (F) para Celsius (C).

Resultado aparece aqui!


Resolução do Código Usado:

let f = parseFloat(prompt("Digite o valor a ser convertido: "))

function conversion() {


let c = (f - 32) * 5/9 // ou 0.5556

console.log("O numero convertido para celcius é: " , c)


}

conversion(f)


Exercicio 2

Escreva uma função que mostre na tela um número fornecido pelo usuário, porém invertido. Por exemplo, o usuário fornece o número 875 e a função mostra na tela o número 578.

Resultado aparece aqui!


Resolução do Código Usado:

let num = prompt("Digite Um numero maior que 100: ")

function inverso(num) {


let numInv = 0

while (num != 0) {


numInv = numInv * 10 + (num % 10)

num = parseInt(num / 10)


}

console.log("O numero inverso do numero digitado é: ",numInv)


}

inverso(num)


Exercicio 3

Escreva uma função que permita contar o número de vogais contidas em uma string fornecida pelo usuário. Por exemplo, o usuário informa a string “Beterraba”, e a função retorna o número 4 (há 4 vogais nessa palavra).

Resultado aparece aqui!


Resolução do Código Usado:

let palavra = String(prompt("Digite uma palavra: "))

function vogais(palavra) {


let i, cont = 0

for (i = 0; i <= palavra.length; i++) {


if (palavra.charAt(i) == "a" || palavra.charAt(i) == "e" || palavra.charAt(i) == "i" || palavra.charAt(i) == "o" || palavra.charAt(i) == "u") {

cont++


}

}

console.log("A quantidade de vogais é", cont)


}

vogais(palavra)

Exercicio 5

Escreva uma função que verifica se um número fornecido pelo usuário em um prompt é primo ou não.

Resultado aparece aqui!


Resolução do Código Usado:

let num = prompt("Digite um numero: ")

function primo(num) {


let i, div = 0, ePrimo, naoPrimo

for (i = 0; i <= num; i++) {


if (num % i == 0) {

div++


if (div == 2) {


ePrimo = "É primo"

console.log("é primo")

} else {


naoPrimo = "Não é primo"

console.log("não é primo")

}

}

}


}

primo(num)


Exercicio 6

Calcular a soma dos números impares maiores que 10 menores que 30

Resultado aparece aqui!


Resolução do Código Usado:

function somaImpar() {


let i, somaImpar = 0

for (i = 11; i <= 30; i++) {


if (i % 2 == 1) {

somaImpar += i


}


}

console.log(somaImpar)


}
